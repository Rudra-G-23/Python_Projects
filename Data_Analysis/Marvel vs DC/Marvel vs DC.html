def dc_or_Marvel (title):
    for keyword in marvel_most_frequent_keywords:
        if keyword in title:
            return 'Marvel'
    for keyword in dc_most_frequent_keywords:
        if keyword in title:
            return 'DC'
    return 'Unknown'
